angular.module('ngAppDemo', ['bbc.sort']).controller('NgAppDemoCtrl', function ($scope) { $scope.sortOpts = {'name': -1}; $scope.items = [ { name: 'John Doe', city: 'New York', country: 'USA' }, { name: 'Tina Tester', city: 'Leipzig', country: 'Germany' }, { name: 'Sam Sample', city: 'Sydney', country: 'Australia' }, { name: 'Max Mustermann', city: 'Toronto', country: 'Kanada' } ]; $scope.sort = function (sort) { $scope.sortOpts = sort; var key = ''; for (var propName in sort) { key = propName; break; } $scope.items.sort(function(a, b) { var x = a[key]; var y = b[key]; return sort[key] === -1 ? (x < y) : (x > y); }); }; $scope.sort($scope.sortOpts); });
